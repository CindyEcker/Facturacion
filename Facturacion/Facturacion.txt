Create database Facturacion 
go

use Facturacion
go

Create table Articulos
(
  ID int primary key Identity,
  Descripcion varchar (50),
  Precio_Unitario  DECIMAL(18, 2),
  EstadoID int
)
go

Create table Clientes
(
  ID int primary key Identity, 
  Nombre_Comercial varchar(100),
  RNCOCEDULA varchar(13),
  Cuenta_contable int, 
  EstadoID int
)

go

Create table vendedores 
(
  ID int primary key Identity,
  Nombre varchar(50),
  Porc_Comision int,
  ID_ESTADO int
)
go


Create table Facturas
(
  ID int primary key Identity, 
  VendedorID int,
  ClienteID int, 
  Fecha datetime,
  Comentario Text,
  ArticuloID int,
  Cantidad int,
  Precio DECIMAL(18, 2) Default 0,
)

go

Create table Asiento_Contable
(
  ID int primary key Identity, 
  Descripcion varchar(50),
  ClienteID int,
  Tipo_Movimiento Char(5),
  Fecha_Asiento Date,
  Monto_Asiento  DECIMAL(18, 2),
  EstadoID int
)
go

CREATE TABLE Estados (
  ID int primary key Identity, 
  Descripcion varchar(50)
)

INSERT INTO ESTADOS (Descripcion) VALUES ('Activo'), ('Inactivo')
